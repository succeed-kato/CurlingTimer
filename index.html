import cv2 # カメラの準備 cap = cv2.VideoCapture(0) print("カメラを起動しています... 終了するには画面を閉じるかアプリを止めてください") while True: ret, frame = cap.read() if not ret: break # 画面の中央に「検知用の四角」を表示してみる（テスト用） height, width, _ = frame.shape cv2.rectangle(frame, (width//4, height//4), (width*3//4, height*3//4), (0, 255, 0), 2) cv2.imshow('Curling Sensor Test', frame) if cv2.waitKey(1) & 0xFF == ord('q'): break cap.release() cv2.destroyAllWindows()
